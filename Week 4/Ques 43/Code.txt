#include <Keypad.h>
#include <LiquidCrystal_I2C.h>
#include <ESP32Servo.h>

const byte ROWS = 4, COLS = 4;
char keys[ROWS][COLS] = {

  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};
byte rowPins[ROWS] = {13, 12, 14, 27};
byte colPins[COLS] = {26, 25, 33, 32};
Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);

LiquidCrystal_I2C lcd(0x27, 16, 2);

Servo doorServo;
#define SERVO_PIN 18
#define LOCKED_ANGLE    0
#define UNLOCKED_ANGLE  90

#define BT_SERIAL Serial2

enum State { WAIT_ID, WAIT_OTP };
State state = WAIT_ID;

String enteredID  = "";
String enteredOTP = "";
String correctOTP = "";

unsigned long otpSentTime = 0;
#define OTP_TIMEOUT 30000

String generateOTP() {

  String otp = "";
  for (int i = 0; i < 6; i++) {

    otp += String(random(0, 10));

  }
  return otp;

}

void setup() {

  Serial.begin(115200);
  BT_SERIAL.begin(9600);

  randomSeed(analogRead(0));

  lcd.init();
  lcd.backlight();

  doorServo.attach(SERVO_PIN);
  doorServo.write(LOCKED_ANGLE);

  showIDScreen();

}

void showIDScreen() {

  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Enter 4-Digit ID");
  lcd.setCursor(0, 1);
  lcd.print("ID: ");

}

void showOTPScreen() {

  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Enter OTP (30s)");
  lcd.setCursor(0, 1);
  lcd.print("OTP: ");

}

void showCountdown(int seconds) {

  lcd.setCursor(10, 0);
  lcd.print("    ");
  lcd.setCursor(10, 0);
  lcd.print(String(seconds) + "s  ");

}

void showResult(bool success) {

  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print(success ? "   ACCESS GRANTED" : "  ACCESS DENIED");
  lcd.setCursor(0, 1);
  lcd.print(success ? "Welcome!" : "Try Again.");

}

void loop() {

  char key = keypad.getKey();

  if (state == WAIT_OTP) {

    unsigned long elapsed = millis() - otpSentTime;
    int remaining = (OTP_TIMEOUT - elapsed) / 1000;

    if (elapsed >= OTP_TIMEOUT) {

      Serial.println("[LOG] OTP expired, ACCESS DENIED");
      showResult(false);
      delay(2000);
      state = WAIT_ID;
      enteredID  = "";
      enteredOTP = "";
      showIDScreen();
      return;

    }
    showCountdown(remaining);

  }

  if (!key) return;

  if (state == WAIT_ID) {

    if (key >= '0' && key <= '9') {

      enteredID += key;
      lcd.setCursor(4 + enteredID.length() - 1, 1);
      lcd.print("*");

      if (enteredID.length() == 4) {

        correctOTP = generateOTP();

        Serial.println("[LOG] ID entered: " + enteredID);
        Serial.println("[LOG] OTP generated: " + correctOTP);

        BT_SERIAL.println("Your OTP: " + correctOTP + " (valid 30s)");

        state = WAIT_OTP;
        otpSentTime = millis();
        enteredOTP = "";
        showOTPScreen();

      }
    } else if (key == '*') {

      if (enteredID.length() > 0) {

        enteredID.remove(enteredID.length() - 1);
        lcd.setCursor(4 + enteredID.length(), 1);
        lcd.print(" ");

      }

    }

  }

  else if (state == WAIT_OTP) {

    if (key >= '0' && key <= '9') {

      enteredOTP += key;
      lcd.setCursor(5 + enteredOTP.length() - 1, 1);
      lcd.print(key);

      if (enteredOTP.length() == 6) {

        bool success = (enteredOTP == correctOTP);

        Serial.println("[LOG] OTP entered: " + enteredOTP + " | Expected: " + correctOTP
                     + " | Result: " + (success ? "SUCCESS" : "FAILURE")
                     + " | ID: " + enteredID);

        showResult(success);

        if (success) {

          doorServo.write(UNLOCKED_ANGLE);
          delay(5000);
          doorServo.write(LOCKED_ANGLE);

        }

        delay(2000);
        state = WAIT_ID;
        enteredID  = "";
        enteredOTP = "";
        showIDScreen();

      }
    } else if (key == '*') {

      if (enteredOTP.length() > 0) {

        enteredOTP.remove(enteredOTP.length() - 1);
        lcd.setCursor(5 + enteredOTP.length(), 1);
        lcd.print(" ");

      }

    }

  }

}
