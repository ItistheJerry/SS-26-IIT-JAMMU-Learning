#include <DHT.h>

#define MQ2_PIN       34
#define DHT_PIN        4
#define DHT_TYPE  DHT11

#define LED_GREEN     27
#define LED_YELLOW    26
#define LED_RED       25
#define BUZZER_PIN    32

#define SAFE_THRESHOLD      300
#define MODERATE_THRESHOLD  600

#define BT_SERIAL Serial2

DHT dht(DHT_PIN, DHT_TYPE);

unsigned long lastLog      = 0;
unsigned long lastBuzz     = 0;
unsigned long logInterval  = 1000;
int           buzzCount    = 0;
int           buzzTarget   = 0;
bool          buzzState    = false;
int           currentLevel = 0;

void setup() {

  Serial.begin(115200);
  BT_SERIAL.begin(9600);
  dht.begin();

  pinMode(LED_GREEN,  OUTPUT);
  pinMode(LED_YELLOW, OUTPUT);
  pinMode(LED_RED,    OUTPUT);
  pinMode(BUZZER_PIN, OUTPUT);

  Serial.println("Timestamp, MQ-2 Value, DHT11 Temp, DHT11 Humidity, Severity Level");

}

void setLED(int level) {

  digitalWrite(LED_GREEN,  level == 0 ? HIGH : LOW);
  digitalWrite(LED_YELLOW, level == 1 ? HIGH : LOW);
  digitalWrite(LED_RED,    level == 2 ? HIGH : LOW);

}

void updateBuzzer(int level) {

  unsigned long now = millis();

  if (level == 0) {

    digitalWrite(BUZZER_PIN, LOW);
    return;

  }

  if (level == 1) {

    if (now - lastBuzz > 3000) {

      lastBuzz   = now;
      buzzCount  = 0;
      buzzTarget = 2;

    }
    if (buzzCount < buzzTarget) {

      if (now - lastBuzz < 200)       { digitalWrite(BUZZER_PIN, HIGH); }
      else if (now - lastBuzz < 400)  { digitalWrite(BUZZER_PIN, LOW);  }
      else if (now - lastBuzz < 600)  { digitalWrite(BUZZER_PIN, HIGH); }
      else                            { digitalWrite(BUZZER_PIN, LOW); buzzCount = buzzTarget; }

    }

  }

  if (level == 2) {

    if (now - lastBuzz > 300) {

      lastBuzz = now;
      buzzState = !buzzState;
      digitalWrite(BUZZER_PIN, buzzState ? HIGH : LOW);

    }

  }

}

void sendBluetoothAlert(int mq2, float temp, float hum, int level) {

  if (level == 0) {

    BT_SERIAL.println("SAFE|Gas=" + String(mq2) + "|Temp=" + String(temp, 1) + "C|Humidity=" + String(hum, 0) + "%");
  } else if (level == 1) {

    BT_SERIAL.println("WARNING|Moderate Smoke Detected|Gas=" + String(mq2) + "|Check Fireplace");
  } else {

    BT_SERIAL.println("DANGER!!|Smoke Level Critical|Gas=" + String(mq2) + "|Open Windows|Leave Room if Necessary");

  }

}

void loop() {

  unsigned long now = millis();

  if (now - lastLog >= logInterval) {

    lastLog = now;

    int   mq2  = analogRead(MQ2_PIN);
    float temp = dht.readTemperature();
    float hum  = dht.readHumidity();

    if (isnan(temp) || isnan(hum)) {

      Serial.println("DHT11 read error!");
      return;

    }

    String level;
    if (mq2 < SAFE_THRESHOLD) {

      currentLevel = 0; level = "SAFE";
    } else if (mq2 < MODERATE_THRESHOLD) {

      currentLevel = 1; level = "MODERATE";
    } else {

      currentLevel = 2; level = "DANGER";

    }

    unsigned long secs = now / 1000;
    Serial.printf("%02lu:%02lu, %d ppm, %.1fC, %.0f%%, %s\n",
                  secs / 60, secs % 60, mq2, temp, hum, level.c_str());

    setLED(currentLevel);
    sendBluetoothAlert(mq2, temp, hum, currentLevel);

  }

  updateBuzzer(currentLevel);

}
